        void Fill_ComboBox_ConstantsBlock2()
        {
            if (Conf != null)
            {
                ListStore dataSource = new ListStore(typeof(string), typeof(string));

                foreach (ConfigurationConstantsBlock block in Conf.ConstantsBlock.Values)
                {
                    dataSource.AppendValues(block.Desc, block.BlockName);
                }

                dataSource.AppendValues("Goalkeeper", 1);
                dataSource.AppendValues("Sweeper", 2);
                dataSource.AppendValues("Defender", 4);
                dataSource.AppendValues("Defensive Mid", 8);

                ComboBox cb = new ComboBox();
                CellRendererText text = new Gtk.CellRendererText();
                cb.Model = dataSource;
                cb.PackStart(text, false);
